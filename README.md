# HbutClassSys

DataBase:MySQL5.7
DataBase Location:
	jdbc.driverClassName=com.mysql.jdbc.Driver
	jdbc.url=jdbc\:mysql\://localhost\:3306/hbut_class_database?useUnicode\\\=true&characterEncoding\\\=utf-8&autoReconnect\\\=true
